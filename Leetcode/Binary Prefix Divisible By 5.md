```cpp
class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& nums) {
        vector<bool> ans(nums.size());
        int num = 0;
        for (int i = 0; i < nums.size(); ++i) {
            num = ((num << 1) | nums[i]) % 5;
            ans[i] = (num == 0);
        }
        return ans;
    }
};
```
**Explanation:**
	Цель: Дан массив из нулей и едениц, каждое число равняет чсилу в диапозоне от 0 до i, переведенное в десятичное число. Нужно проверить каждое ичсло на его деление на пять без остатка.
	Решение: С помощью битовых операций накапливать число - сдвиганием на лево на один(или уножение на 2), добавление в конце еденички с помощью побитовой операции или(или прибавление nums\[i]) и взятием модуля по 5, т.к. чтобы избежать переполнения и нужно знать только остаток.
